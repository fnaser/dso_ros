# Calibration

rosrun camera_calibration cameracalibrator.py --size 10x7 --square 0.0229 image:=/usb_cam/image_raw camera:=/usb_cam/ #count edges

ROS_NAMESPACE=usb_cam rosrun image_proc image_proc

Edit the PtamFixParams.yaml file and set the parameters "ImageSizeX" "ImageSizeY" to your image resolution.

Edit the cameracalibrator.launch file to map the image to your image topic.

roslaunch ptam cameracalibrator.launch
